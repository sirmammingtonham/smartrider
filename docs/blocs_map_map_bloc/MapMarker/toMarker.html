dynamic toMarker ()

Implementation

Marker toMarker() => Marker(
    markerId: MarkerId(id),
    position: LatLng(
      position.latitude,
      position.longitude,
    ),
    infoWindow: InfoWindow(title: this.info),
    icon: icon,
    onTap: () {
      controller.animateCamera(
        CameraUpdate.newCameraPosition(CameraPosition(
            target: LatLng(
              position.latitude,
              position.longitude,
            ),
            zoom: 18,
            tilt: 50)),
      );
    });
