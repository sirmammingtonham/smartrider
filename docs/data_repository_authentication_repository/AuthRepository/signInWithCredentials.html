Future signInWithCredentials(String email, String password) async {
  try {
    return await _firebaseAuth.signInWithEmailAndPassword(
      email: email.trim(),
      password: password,
    );
  } on PlatformException catch (e) {
    print(e);
    return e;
  }
}
