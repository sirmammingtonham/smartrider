Map<String, dynamic> toJson() {
  final Map<String, dynamic> data = new Map<String, dynamic>();
  data['id'] = this.id;
  data['name'] = this.name;
  data['created'] = this.created;
  data['updated'] = this.updated;
  data['enabled'] = this.enabled;
  data['tracker_id'] = this.trackerId;
  return data;
